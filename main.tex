\documentclass[phd, ist, showframe]{VISTEC}

%% Load thesisinfo.tex
\input{thesisinfo} %% edit title, thesis approval, and committee at `thesisinfo.tex`

%% Load bibliography file
\addbibresource{bibliography.bib}

%%%%%%%%%%%%%%%%%%%%%% Add your custom packages %%%%%%%%%%%%%%%%%%%%%

\usepackage{multirow}
\usepackage[version=4]{mhchem}
\usepackage{chemfig}

\begin{document} 

    %% Set global line spacing
    \singlespacingplus % 1.5x line spacing

    %% Title and approval pages
    \maketitle
    \makeapproval

    %%%%%%%%%%%%%%%%%%%%%%%%%%%% Prologue %%%%%%%%%%%%%%%%%%%%%%%%%%%% 
    %% Roman page numbering
    \fancyhf[cf]{\thepage}
    \pagenumbering{roman}
    \setcounter{page}{2}

    \abstract                       %% Content at `contents/abstract.tex`
    \acknowledgment                 %% Content at `contents/acknowledgment.tex`

    \listoftables
    \listoffigures
    \abbreviations                  %% Content at `contents/abbreviations.tex`
    %%%%%%%%%%%%%%%%%%%%%%%%%%%% Main Chapters %%%%%%%%%%%%%%%%%%%%%%%%%%%% 
    %% Arabic page numbering
    \pagenumbering{arabic}
    \setcounter{page}{1}
    
    \include{contents/chapter1}
    \include{contents/chapter2}
    \include{contents/chapter3}
    \include{contents/chapter4}
    \include{contents/chapter5}
    
    %%%%%%%%%%%%%%%%%%%%%%%%%%%% Epilogue %%%%%%%%%%%%%%%%%%%%%%%%%%%% 
    %% References
    \vancouverbib

    %% Appendices
    \include{contents/appendix}   %% Content at `contents/appendix.tex`

    %% Author Biography
    \authorbiography                %% Content at `contents/authorbiography.tex`
